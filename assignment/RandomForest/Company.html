# splitting the target Variable in to 2 parts
def split(num):
    if num < 5:
        return 'Low sale'
    elif num >= 5 and num < 10:
        return 'Average sale'
    else:
        return 'High sale'
company_data.Sales = company_data.Sales.apply(split)
